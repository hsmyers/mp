427  float ER2 = 1089.0;

429  float zOld;
430 
431  Zx2 = Zx * Zx;
432  Zy2 = Zy * Zy;
433  absZ2 = Zx * Zx + Zy * Zy;
434  absdZ2 = dZx * dZx + dZy * dZy;
435 
436  for ( cp.n = 0; cp.n < iMax; cp.n++ ) {
437  zOld = absZ2;
438  if ( absZ2 > ER2 ) {
439  break;
440  }

442  dZy = 2 * ( Zx * dZy + Zy * dZx );
443  dZx = temp;

447

449  Zy2 = Zy * Zy;
450  absZ2 = Zx2 + Zy2;
451  absdZ2 = dZx * dZx + dZy * dZy;
