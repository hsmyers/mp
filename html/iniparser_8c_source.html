658 static line_status iniparser_line(
659  const char * input_line,
660  char * section,
661  char * key,
662  char * value)
663 {
664  line_status sta ;
665  char * line = NULL;
666  size_t len ;
667 
668  line = xstrdup(input_line);
669  len = strstrip(line);
670 
671  sta = LINE_UNPROCESSED ;
672  if (len<1) {
673  /* Empty line */
674  sta = LINE_EMPTY ;
675  } else if (line[0]=='#' || line[0]==';') {
676  /* Comment line */
677  sta = LINE_COMMENT ;
678  } else if (line[0]=='[' && line[len-1]==']') {
679  /* Section name */
680  sscanf(line, "[%[^]]", section);
681  strstrip(section);
682  strlwc(section, section, len);
683  sta = LINE_SECTION ;
684  } else if (sscanf (line, "%[^=] = \"%[^\"]\"", key, value) == 2
685  || sscanf (line, "%[^=] = '%[^\']'", key, value) == 2) {
686  /* Usual key=value with quotes, with or without comments */
687  strstrip(key);
688  strlwc(key, key, len);
689  /* Don't strip spaces from values surrounded with quotes */
690  sta = LINE_VALUE ;
691  } else if (sscanf (line, "%[^=] = %[^;#]", key, value) == 2) {
692  /* Usual key=value without quotes, with or without comments */
693  strstrip(key);
694  strlwc(key, key, len);
695  strstrip(value);
696  /*
697  * sscanf cannot handle '' or "" as empty values
698  * this is done here
699  */
700  if (!strcmp(value, "\"\"") || (!strcmp(value, "''"))) {
701  value[0]=0 ;
702  }
703  sta = LINE_VALUE ;
704  } else if (sscanf(line, "%[^=] = %[;#]", key, value)==2
705  || sscanf(line, "%[^=] %[=]", key, value) == 2) {
706  /*
707  * Special cases:
708  * key=
709  * key=;
710  * key=#
711  */
712  strstrip(key);
713  strlwc(key, key, len);
714  value[0]=0 ;
715  sta = LINE_VALUE ;
716  } else {
717  /* Generate syntax error */
718  sta = LINE_ERROR ;
719  }
720 
721  free(line);
722  return sta ;
723 }

739 dictionary * iniparser_load(const char * ininame)
740 {
741  FILE * in ;
742 
743  char line [ASCIILINESZ+1] ;
744  char section [ASCIILINESZ+1] ;
745  char key [ASCIILINESZ+1] ;
746  char tmp [(ASCIILINESZ * 2) + 1] ;
747  char val [ASCIILINESZ+1] ;
748 
749  int last=0 ;
750  int len ;
751  int lineno=0 ;
752  int errs=0;
753  int mem_err=0;
754 
755  dictionary * dict ;
756 
757  if ((in=fopen(ininame, "r"))==NULL) {
758  iniparser_error_callback("iniparser: cannot open %s\n", ininame);
759  return NULL ;
760  }
761 
762  dict = dictionary_new(0) ;
763  if (!dict) {
764  fclose(in);
765  return NULL ;
766  }
767 
768  memset(line, 0, ASCIILINESZ);
769  memset(section, 0, ASCIILINESZ);
770  memset(key, 0, ASCIILINESZ);
771  memset(val, 0, ASCIILINESZ);
772  last=0 ;
773 
774  while (fgets(line+last, ASCIILINESZ-last, in)!=NULL) {
775  lineno++ ;
776  len = (int)strlen(line)-1;
777  if (len<=0)
778  continue;
779  /* Safety check against buffer overflows */
780  if (line[len]!='\n' && !feof(in)) {
781  iniparser_error_callback(
782  "iniparser: input line too long in %s (%d)\n",
783  ininame,
784  lineno);
785  dictionary_del(dict);
786  fclose(in);
787  return NULL ;
788  }
789  /* Get rid of \n and spaces at end of line */
790  while ((len>=0) &&
791  ((line[len]=='\n') || (isspace(line[len])))) {
792  line[len]=0 ;
793  len-- ;
794  }
795  if (len < 0) { /* Line was entirely \n and/or spaces */
796  len = 0;
797  }
798  /* Detect multi-line */
799  if (line[len]=='\\') {
800  /* Multi-line value */
801  last=len ;
802  continue ;
803  } else {
804  last=0 ;
805  }
806  switch (iniparser_line(line, section, key, val)) {
807  case LINE_EMPTY:
808  case LINE_COMMENT:
809  break ;
810 
811  case LINE_SECTION:
812  mem_err = dictionary_set(dict, section, NULL);
813  break ;
814 
815  case LINE_VALUE:
816  sprintf(tmp, "%s:%s", section, key);
817  mem_err = dictionary_set(dict, tmp, val);
818  break ;
819 
820  case LINE_ERROR:
821  iniparser_error_callback(
822  "iniparser: syntax error in %s (%d):\n-> %s\n",
823  ininame,
824  lineno,
825  line);
826  errs++ ;
827  break;
828 
829  default:
830  break ;
831  }
832  memset(line, 0, ASCIILINESZ);
833  last=0;
834  if (mem_err<0) {
835  iniparser_error_callback("iniparser: memory allocation failure\n");
836  break ;
837  }
838  }
839  if (errs) {
840  dictionary_del(dict);
841  dict = NULL ;
842  }
843  fclose(in);
844  return dict ;
845 }
